// ✅ GOOD: Handling App Links in Activity
override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    handleIntent(intent)
}

private fun handleIntent(intent: Intent) {
    val appLinkData: Uri? = intent.data
    if (Intent.ACTION_VIEW == intent.action && appLinkData != null) {
        val path = appLinkData.lastPathSegment
        // Route to specific screen
        navController.navigate("product/$path")
    }
}

// ✅ GOOD: Dynamic Sharing with Chooser
fun shareProduct(context: Context, product: Product) {
    val sendIntent: Intent = Intent().apply {
        action = Intent.ACTION_SEND
        putExtra(Intent.EXTRA_TEXT, "Check out ${product.name}: ${product.link}")
        type = "text/plain"
    }
    val shareIntent = Intent.createChooser(sendIntent, null)
    context.startActivity(shareIntent)
}

// ✅ GOOD: In-App Review Flow
fun requestReview(activity: Activity) {
    val manager = ReviewManagerFactory.create(activity)
    val request = manager.requestReviewFlow()
    request.addOnCompleteListener { task ->
        if (task.isSuccessful) {
            val reviewInfo = task.result
            val flow = manager.launchReviewFlow(activity, reviewInfo)
            // ...
        }
    }
}

**Boundaries**
✅ Always do:
- Run `./gradlew lint`
- Handle Deep Links in all states (Cold start, Background)
- Add `meta-data` to AndroidManifest.xml for App Indexing
- Resize images for Play Store Screenshots automatically
- Keep changes under 50 lines
- Test Deep Links via `adb shell am start`

⚠️ Ask first:
- Changing the App Name or Application ID (Package Name)
- Modifying the Launcher Icon (requires Adaptive Icon work)
- Implementing "Incentivized Installs" (banned by Google)
- Sending "Promotional" Push Notifications (requires explicit channel)

🚫 Never do:
- Fake reviews or ratings
- Use "Clickbait" titles that violate Play Store Metadata policies
- Break the Deep Link schema (backward compatibility)
- Ask for review immediately after a crash or error

- Notifications are a privilege, not a right: don't abuse them
